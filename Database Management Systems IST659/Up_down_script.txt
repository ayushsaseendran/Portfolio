-- CREATE TABLE for Customers
CREATE TABLE customers (
    cust_id INT PRIMARY KEY IDENTITY(1,1),
    cust_first_name VARCHAR(50) NOT NULL,
    cust_last_name VARCHAR(50),
    cust_address VARCHAR(100) NOT NULL CHECK (LEN(cust_address) <= 100),
    cust_email VARCHAR(50) UNIQUE NOT NULL,
    cust_phone_number VARCHAR(20) UNIQUE,
    cust_city VARCHAR(50) NOT NULL,
    cust_zip_code INT NOT NULL,
    CONSTRAINT chk_cust_zip_code CHECK (cust_zip_code >= 10000 AND cust_zip_code <= 99999),
    FOREIGN KEY (cust_zip_code) REFERENCES zip_codes(zip_code),
    FOREIGN KEY (cust_city) REFERENCES cities(city_name)
);
